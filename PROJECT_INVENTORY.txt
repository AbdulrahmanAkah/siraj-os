import json
from datetime import datetime
from graphlib import TopologicalSorter, CycleError


REPORTS_DIR = Path("reports")
REPORTS_DIR.mkdir(exist_ok=True)


def save_report(name: str, data):
    path = REPORTS_DIR / name

    with open(path, "w", encoding="utf8") as f:
        json.dump(
            data,
            f,
            ensure_ascii=False,
            indent=4,
            default=str
        )


def build_dependency_graph(modules: dict[str, ModuleInfo]):

    graph = {}

    for module_name, info in modules.items():
        graph[module_name] = sorted(info.imports)

    return graph


def detect_cycles(graph):

    try:
        ts = TopologicalSorter(graph)
        list(ts.static_order())
        return []

    except CycleError as e:
        return list(e.args)

def find_merge_candidates(classes):

    result = {}

    for cls, files in classes.items():

        if len(files) < 2:
            continue

        canonical = sorted(files, key=len)[0]

        result[cls] = {
            "canonical": canonical,
            "duplicates": sorted(files)
        }

    return result
